# LinuxCommands
##1.File System Commands
```commandline
# BIOS = Basic Input Output System
1. ls -a             # list with hidden files
2. ls -la            # list with long and hidden
3. cd <dir_name> || cd .. || cd ../..
4. ls <dir_name> || ls -la || ls -lah[with size]
5. only cd or cd ~ --> home dir

6. mkdir <dir_name> # create directory 
7. touch <filename.extention>    # create file
8. nano <filename.extention>     # create file and open with nano
9. cp <file> <cpyFile>           # replicate a file

10. mv <file> <renamed>     # rename file with move
11. mv <file> <dir>         # move file to dir
12. mv <file> ../rename.ext

13. rm <file>             # delete file permanent
14. rm -d <dir>           # delete empty dir
15. cp -R <dir>/ <cpydir> # copy dir
16. mv <dir> <renameDir>  # rename dir
17. rm -rf <dir>          # force remove including file

18. find .                # showing all the file in this dir
19. find <dir_location>
20. find . -type d        # all the dir and subdir
21. find . -type f        # all the file of dir and subdir
```
